sample_csv = (

    df = pd.read_csv("/mnt/data/employee_data.csv")

    df = pd.read_csv(io.StringIO(sample_csv))

# 1) Frequency count for "HR" department
hr_count = (df["department"] == "HR").sum()

# 2) Histogram-style bar chart of department distribution
dept_counts = df["department"].value_counts().sort_index()

plt.figure()
dept_counts.plot(kind="bar")
plt.title("Department Distribution")
plt.xlabel("Department")
plt.ylabel("Count")
plt.tight_layout()
plt.savefig("department_distribution.png")
